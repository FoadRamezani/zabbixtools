install offline zabbix agent :
		1 - extract zabbix agent file
		2 - cp -v sbin/* /usr/sbin/
		3 - cp -v bin/* /usr/bin/*
		4 - mkdir /etc/zabbix
		5 - cp -v conf/zabbix_agentd.conf /etc/zabbix
		6 - mkdir /etc/zabbix/zabbix_agentd.d
		7 - cp -v conf/zabbix_agentd/* /etc/zabbix/zabbix_agentd.d/
		8 - vi /etc/zabbix/zabbix_agentd.conf
			8.1 - Server=192.168.1.100
			8.2 - ServerActive=192.168.1.100
			8.3 - Hostname=offline.local
		9 - useradd zabbix
		10 - zabbix_agentd -c /etc/zabbix/zabbix_agentd.conf
		11 - 
		11 - firewall-cmd --add-port=10050/tcp --permanent
		12 - firewall-cmd --reload
		13 - vi /etc/systemd/system/zabbix_agentd.service	

			[Unit]
			Description=Zabbix Agent Daemon
			After=syslog.target network.target

			[Service]
			Type=simple
			ExecStart=/usr/sbin/zabbix_agentd -c /etc/zabbix/zabbix_agentd.conf 

			[Install]
			WantedBy=multi-user.target
		14 - chmod 777 zabbix_agentd.service
		15 - mkdir /run/zabbix
		16 - 
		17 -
